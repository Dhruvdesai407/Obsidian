    ```javascript
        });
      });
    };

    const writeFilePromise = (filePath, data, encoding) => {
      return new Promise((resolve, reject) => {
        fs.writeFile(filePath, data, encoding, (err) => {
          if (err) reject(err);
          else resolve(); // âœ… Fulfill (no data needed for write success)
        });
      });
    };

    console.log('Initiating a sequence of operations with Promises...');

    readFilePromise('file1.txt', 'utf8') // 1ï¸âƒ£ Read file1 (returns a Promise)
      .then(data1 => { // ğŸ‰ When file1 is read successfully...
        console.log('âœ… Read file1 (Promise):', data1);
        return writeFilePromise('file2.txt', `Processed: ${data1}`, 'utf8'); // 2ï¸âƒ£ Chain: Write to file2
      })
      .then(() => { // ğŸ‰ When file2 is written successfully...
        console.log('âœ… Wrote to file2 (Promise).');
        return readFilePromise('file2.txt', 'utf8'); // 3ï¸âƒ£ Chain: Read file2 again
      })
      .then(data2 => { // ğŸ‰ When file2 is read again successfully...
        console.log('âœ… Read file2 again (Promise):', data2);
        console.log('Sequence complete with Promises!'); âœ¨
      })
      .catch(err => { // ğŸš¨ CATCH ALL errors in the chain!
        console.error('âŒ An error occurred in the Promise chain:', err);
      });
    ```
      * **Why it's cool:** *Flattens asynchronous code, making it far more readable and manageable.* The `.then().catch()` pattern is clean\!

### GraphQL with Node.js (Modern APIs\! ğŸ“¡)

  * **Definition:** A *query language for your API*, and a runtime for fulfilling those queries with your existing data. It's an *alternative to REST*.

  * **Core Concepts:**
      * **Schema Definition Language (SDL):** Defines your data types and what operations (queries, mutations) are available.
      * **Resolvers:** Functions that *fetch the actual data* for your defined types and operations.
